const WORKER_URL = "https://pysiga.d-alexjose.workers.dev"; const email = new URLSearchParams(window.location.search).get('email'); let datosGlobales = null; let chartEvolucion = null; let chartDistribucion = null; let chartPorLote = null; document.addEventListener('DOMContentLoaded', () => { if (!email) { alert('❌ No se encontró el correo del usuario'); return; } cargarDatos(); }); async function cargarDatos(){ mostrarLoader(true); try{ const response = await fetch(WORKER_URL,{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({ accion:'reportePesajeCorporal', email:email }) }); if(!response.ok){ throw new Error("Error de servidor"); } const data = await response.json(); if(!data.ok){ throw new Error(data.error || "Error cargando datos"); } datosGlobales = data; poblarFiltros(); aplicarFiltros(); document.getElementById("contenido").style.display="block"; }catch(err){ console.error(err); alert("❌ "+err.message); } mostrarLoader(false); } function aplicarFiltros(){ if(!datosGlobales) return; const loteFiltro = document.getElementById("filtroLote").value; const clasifFiltro = document.getElementById("filtroClasif").value; const metaGanancia = parseFloat(document.getElementById("metaGanancia").value) || 0; let animales = [...(datosGlobales?.listados?.animalesConDatos || [])]; if(loteFiltro){ animales = animales.filter(a=>a.lote===loteFiltro); } if(clasifFiltro){ animales = animales.filter(a=>a.clasificacion===clasifFiltro); } const animalesConGanancia = animales.filter(a=>a.gananciaDiaria !== null && !isNaN(a.gananciaDiaria)); /* KPI GANANCIA GENERAL */ const gananciaPromedio = animalesConGanancia.length ? animalesConGanancia.reduce((s,a)=>s + a.gananciaDiaria ,0) / animalesConGanancia.length : 0; /* KPI KILOS TOTALES */ const kilosTotales = animales.reduce( (s,a)=> s + (a.pesoActual || 0), 0 ); /* KPI META */ const animalesMeta = metaGanancia>0 ? animalesConGanancia.filter(a=>a.gananciaDiaria >= metaGanancia) : []; const metaCumplida = animalesMeta.length; const metaPorcentaje = animalesConGanancia.length ? (metaCumplida / animalesConGanancia.length) * 100 : 0; /* KPI DESMEJORADOS */ const desmejorados = (datosGlobales?.listados?.animalesDesmejorados || []) .filter(a=>{ if(loteFiltro && a.lote!==loteFiltro) return false; if(clasifFiltro && a.clasificacion!==clasifFiltro) return false; return true; }); actualizarKPIs( gananciaPromedio, kilosTotales, animales.length, desmejorados.length, metaCumplida, metaPorcentaje, metaGanancia ); const kpisPorLote = calcularKPIsPorLote(animales); const kpisPorClasif = calcularKPIsPorClasificacion(animales); const evolucion = datosGlobales?.graficos?.evolucionGananciaGeneral || []; const distribucion = calcularDistribucion(animalesConGanancia); renderizarGraficos(evolucion,distribucion,kpisPorLote); renderizarTablas(kpisPorLote,kpisPorClasif,desmejorados,animales,metaGanancia); } function calcularKPIsPorLote(animales){ const lotes = {}; animales.forEach(a=>{ if(!lotes[a.lote]){ lotes[a.lote]={ totalAnimales:0, kilosTotales:0, ganancias:[] }; } lotes[a.lote].totalAnimales++; lotes[a.lote].kilosTotales += (a.pesoActual || 0); if(a.gananciaDiaria!==null){ lotes[a.lote].ganancias.push(a.gananciaDiaria); } }); Object.keys(lotes).forEach(l=>{ const g=lotes[l].ganancias; lotes[l].gananciaDiariaPromedio = g.length ? g.reduce((s,v)=>s+v,0)/g.length : null; }); return lotes; } function calcularKPIsPorClasificacion(animales){ const clasif={}; animales.forEach(a=>{ if(!a.clasificacion) return; if(!clasif[a.clasificacion]){ clasif[a.clasificacion]={ totalAnimales:0, kilosTotales:0, ganancias:[] }; } clasif[a.clasificacion].totalAnimales++; clasif[a.clasificacion].kilosTotales+=(a.pesoActual||0); if(a.gananciaDiaria!==null){ clasif[a.clasificacion].ganancias.push(a.gananciaDiaria); } }); Object.keys(clasif).forEach(c=>{ const g=clasif[c].ganancias; clasif[c].gananciaDiariaPromedio = g.length ? g.reduce((s,v)=>s+v,0)/g.length : null; }); return clasif; } function calcularDistribucion(animales){ const d={ "Negativa":0, "0 a 0.5":0, "0.5 a 1":0, "1 a 1.5":0, "1.5 a 2":0, "Más de 2":0 }; animales.forEach(a=>{ const g=a.gananciaDiaria; if(g<0) d["Negativa"]++; else if(g<0.5) d["0 a 0.5"]++; else if(g<1) d["0.5 a 1"]++; else if(g<1.5) d["1 a 1.5"]++; else if(g<2) d["1.5 a 2"]++; else d["Más de 2"]++; }); return d; } function actualizarKPIs(general,kilos,total,desm,meta,porcentaje,metaValor){ document.getElementById("kpiGananciaGeneral").textContent=general.toFixed(2); document.getElementById("kpiKilosTotales").textContent=Math.round(kilos).toLocaleString(); document.getElementById("kpiTotalAnimales").textContent=total; document.getElementById("kpiDesmejorados").textContent=desm; document.getElementById("kpiMetaCumplida").textContent=meta; document.getElementById("kpiMetaPorcentaje").textContent= `${porcentaje.toFixed(1)}% ${metaValor>0?`(meta ${metaValor})`:''}`; } function cambiarTab(tabId,btn){ document.querySelectorAll(".tab").forEach(t=>t.classList.remove("active")); document.querySelectorAll(".tab-content").forEach(c=>c.classList.remove("active")); btn.classList.add("active"); document.getElementById(`tab-${tabId}`).classList.add("active"); } function mostrarLoader(v){ document.getElementById("loader").style.display=v?"block":"none"; } function recargarDatos(){ cargarDatos(); } function imprimir(){ window.print(); } function cerrar(){ if(window.opener){ window.close(); }else{ window.location.href="/"; } }
